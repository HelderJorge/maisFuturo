struct formandos
{

 float bolsa;
};
main()
{
 setlocale(LC_ALL, "Portuguese");
 char resp = 's';
 do
 {
  int i,num;
  printf_s("Vamos inserir dados de uma formação. Digite o nº de formandos da turma: ");
  scanf_s("%d", &num);
  struct formandos formando[40];
  for (i = 0; i < num; i++)
  {
   getchar();
   printf_s("\nDigite o nome do %dº formando.\n",i+1);

   scanf_s(" %c", &formando[i].e_civil);
   printf_s("Digite o valor da bolsa do %dº formando.\n",i+1);
   scanf_s ("%f", &formando[i].bolsa);

 }while (resp == 's');
