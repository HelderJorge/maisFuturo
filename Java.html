package Lista_Estudantes;
import java.time.LocalDate;
public class Student {
    private String nome, email;
    private LocalDate datanascimento;
    private Double classificacao;
    public String getNome() {
        return this.nome;
    }
    public String getEmail() {
        return this.email;
    }
    public int getAnonascimento() {
        return this.datanascimento.getYear();
    }
    public Double getClassificacao() {
        return this.classificacao;
    }
    public Student(String nome, String email, LocalDate datanascimento, Double classificacao)
    {
        this.nome = nome;
        if (!email.contains("@"))
        {
            throw new IllegalArgumentException("Email inválido" + email);
        }
            this.email = email;
            this.datanascimento = datanascimento;
            this.classificacao = classificacao;
    }
    @Override //anotação para re-escrever o toString da classe Object. Esse metodo returna o hashcode em hexadecimal.
    public String toString ()
    {
        return "Student{" +
                "nome='" + nome + '\'' +
                ", email='" + email + '\'' +
                ", datanascimento=" + datanascimento +
                ", classificacao=" + classificacao +
                '}';
    }
}
